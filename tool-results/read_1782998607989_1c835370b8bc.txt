   207→        store.setHistoryCount(hist.count ?? 0);

   213→        toast.error("Error al cargar datos iniciales");

   236→      const res = await fetch("/api/characters");
   237→      const chars = await res.json();
   238→      store.setCharacters(Array.isArray(chars) ? chars : []);
   239→    } catch {
   240→      /* silent */

   242→  }, [store]);

   255→          name: newName.trim(),
   256→          description: newDesc.trim(),

   340→        method: "POST",

   343→          characterId: selectedChar.id,

   345→          characterDesc: selectedChar.description,

   364→        throw new Error(data.error || "Error en la generación");

   369→      store.setHistoryCount((prev) => prev + 1);
   370→      toast.success(
   371→        `Campaña generada: ${data.ideas.length} propuestas creadas`
   372→      );
   373→    } catch (e) {
   374→      const msg = e instanceof Error ? e.message : "Error desconocido";
   375→      store.setGenerationError(msg);
   376→      toast.error(msg);
   377→    } finally {
   378→      store.setIsGenerating(false);
   379→    }
   380→  };

   382→  const handleRefine = async (index: number) => {

   385→      toast.error("Escribe instrucciones para refinar");

   393→      const res = await fetch("/api/refine", {
   394→        method: "POST",
   395→        headers: { "Content-Type": "application/json" },
   396→        body: JSON.stringify({ idea, instructions, apiKey: store.apiKey, model: store.model }),

   406→      toast.error(
